def chi_square_test(df):
    chi_square_value, p_value, degree_of_freedom, expected_values=stats.chi2_contingency(df)
    print("The chi-square value is :",chi_square_value)
    print("The p value is :",p_value)
    print("The number of degree of freedom is :",degree_of_freedom)
    print("The expected values is :",expected_values)
    return print('No expect values is less than 5, otherwise, the chi-square test is not valid. The minimum of expected values is: ',expected_values.min())
